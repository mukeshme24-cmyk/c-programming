#include<stdio.h>
int main(){
   int arr[100],n,i,mini;
   printf("Enter number of element:");
   scanf("%d",&n);
   printf("Enter %d elements:\n",n);
   for(i=0;i<n;i++){
       scanf("%d",&arr[i]);
   }
   mini=arr[0];
   for(i=1;i<n;i++){
       if(arr[i]>mini)
       mini=arr[i];
   }
   printf("Minimum element=%d\n",mini);
   return 0;
}
