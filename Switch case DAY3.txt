Switch case 
#include<stdio.h>
int main(){
    int choice;
    do{
        printf("\n==menu==\n");
        printf("1.say cricket\n");
        printf("2.say kho-kho\n");
        printf("3.say football\n");
        printf("4.say free fire\n");
        printf("5.exit\n");
        printf("Enter your choice:");
        scanf("%d",&choice);
        switch(choice){
            case 1:
            printf("cricket\n");
            break;
            case 2:
            printf("kho-kho\n");
            break;
            case 3:
            printf("football\n");
            break;
            case 4:
            printf("free fire\n");
            break;
            printf("exit\n");
            break;
            default:
            printf("=Invalid choice\n");
        }
    }while(choice!=4);
    return 0;
}
