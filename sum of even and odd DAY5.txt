#include <stdio.h>

int main() {
    int arr[] = {10, 21, 32, 43, 54, 65};
    int n = sizeof(arr) / sizeof(arr[0]);
    int evenSum = 0, oddSum = 0;

    for (int i = 0; i < n; i++) {
        if (arr[i] % 2 == 0)
            evenSum += arr[i];
        else
            oddSum += arr[i];
    }

    printf("Sum of even numbers: %d\n", evenSum);
    printf("Sum of odd numbers: %d\n", oddSum);

    return 0;
}
